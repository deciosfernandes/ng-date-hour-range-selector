directive on a native <input> · initial range: last 7 days

Apply drsDateRangePicker directly to any existing <input>. The selected range is shown inside the input. When a value matches a predefined range the label is shown instead.

@if (directiveControl.value; as range) {
Selected range:
Start: {{ range.start | date: 'medium' }}
End: {{ range.end | date: 'medium' }}
} 
// component.ts
readonly directiveControl = new FormControl<DateRange | null>(null);
readonly last7DaysRange: PredefinedRange = {
  label: 'Last 7 days',
  range: () => {
    const end = new Date();
    const start = new Date();
    start.setDate(start.getDate() - 6);
    return { start, end };
  },
};

<!-- template -->
<input
  drsDateRangePicker
  [formControl]="directiveControl"
  [initialRange]="last7DaysRange"
/>

custom predefined ranges

// component.ts
readonly customRanges: PredefinedRange[] = [
  {
    label: 'Next 7 days',
    range: () => {
      const start = new Date();
      const end = new Date();
      end.setDate(end.getDate() + 6);
      return { start, end };
    },
  },
  { label: 'Next 30 days', range: () => { /* ... */ } },
  { label: 'Next quarter',  range: () => { /* ... */ } },
];

<!-- template -->
<drs-date-range-picker
  [predefinedRanges]="customRanges"
/>

min / max date boundaries · current month only

// component.ts — restrict to current month
private readonly _now = new Date();
readonly minDate = new Date(this._now.getFullYear(), this._now.getMonth(), 1);
readonly maxDate = new Date(this._now.getFullYear(), this._now.getMonth() + 1, 0);

<!-- template -->
<drs-date-range-picker
  [minDate]="minDate"
  [maxDate]="maxDate"
/>
